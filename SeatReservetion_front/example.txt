    <script>
        const { createApp, ref, onMounted, watch } = Vue;

        createApp({
            setup() {
                const canvas = ref(null);
                const currentTool = ref('external');
                const walls = ref([]);
                const partitions = ref([]);
                const currentLine = ref([]);
                const isDrawing = ref(false);
                const mousePos = ref({ x: 0, y: 0 });
                const gridSize = ref(20);
                const selectedWall = ref(null);

                const ctx = ref(null);
                const canvasWidth = 1200;
                const canvasHeight = 800;

                onMounted(() => {
                    const c = canvas.value;
                    c.width = canvasWidth;
                    c.height = canvasHeight;
                    ctx.value = c.getContext('2d');
                    draw();

                    document.addEventListener('keydown', handleKeyDown);
                });

                const snapToGrid = (value) => {
                    return Math.round(value / gridSize.value) * gridSize.value;
                };

                const getMousePos = (e) => {
                    const rect = canvas.value.getBoundingClientRect();
                    const scaleX = canvasWidth / rect.width;
                    const scaleY = canvasHeight / rect.height;
                    return {
                        x: snapToGrid((e.clientX - rect.left) * scaleX),
                        y: snapToGrid((e.clientY - rect.top) * scaleY)
                    };
                };

                const draw = () => {
                    const c = ctx.value;
                    c.clearRect(0, 0, canvasWidth, canvasHeight);

                    // Draw grid
                    drawGrid();

                    // Draw completed walls
                    walls.value.forEach(wall => {
                        drawWall(wall, '#1e293b', 8);
                    });

                    // Draw completed partitions
                    partitions.value.forEach(part => {
                        drawWall(part, '#94a3b8', 3);
                    });

                    // Draw current line being drawn
                    if (currentLine.value.length > 0 && isDrawing.value) {
                        const color = currentTool.value === 'external' ? '#1e293b' : '#94a3b8';
                        const width = currentTool.value === 'external' ? 8 : 3;
                        
                        // Draw completed segments
                        for (let i = 0; i < currentLine.value.length - 1; i++) {
                            drawLineSegment(
                                currentLine.value[i],
                                currentLine.value[i + 1],
                                color,
                                width
                            );
                        }

                        // Draw preview line to mouse
                        drawLineSegment(
                            currentLine.value[currentLine.value.length - 1],
                            mousePos.value,
                            color,
                            width,
                            true
                        );
                    }

                    // Draw selection
                    if (selectedWall.value) {
                        drawSelection(selectedWall.value);
                    }
                };

                const drawGrid = () => {
                    const c = ctx.value;
                    c.strokeStyle = '#e2e8f0';
                    c.lineWidth = 1;

                    for (let x = 0; x <= canvasWidth; x += gridSize.value) {
                        c.beginPath();
                        c.moveTo(x, 0);
                        c.lineTo(x, canvasHeight);
                        c.stroke();
                    }

                    for (let y = 0; y <= canvasHeight; y += gridSize.value) {
                        c.beginPath();
                        c.moveTo(0, y);
                        c.lineTo(canvasWidth, y);
                        c.stroke();
                    }
                };

                const drawWall = (wall, color, width) => {
                    for (let i = 0; i < wall.points.length - 1; i++) {
                        drawLineSegment(wall.points[i], wall.points[i + 1], color, width);
                    }
                };

                const drawLineSegment = (start, end, color, width, isPreview = false) => {
                    const c = ctx.value;
                    c.beginPath();
                    c.moveTo(start.x, start.y);
                    c.lineTo(end.x, end.y);
                    c.strokeStyle = color;
                    c.lineWidth = width;
                    c.lineCap = 'round';
                    c.lineJoin = 'round';
                    if (isPreview) {
                        c.setLineDash([5, 5]);
                        c.globalAlpha = 0.6;
                    }
                    c.stroke();
                    c.setLineDash([]);
                    c.globalAlpha = 1;
                };

                const drawSelection = (wall) => {
                    const c = ctx.value;
                    c.strokeStyle = '#667eea';
                    c.lineWidth = 2;
                    c.setLineDash([5, 5]);
                    
                    wall.points.forEach(point => {
                        c.beginPath();
                        c.arc(point.x, point.y, 8, 0, Math.PI * 2);
                        c.stroke();
                    });
                    
                    c.setLineDash([]);
                };

                const handleMouseDown = (e) => {
                    if (currentTool.value === 'select') {
                        selectWall(e);
                        return;
                    }

                    const pos = getMousePos(e);
                    
                    if (!isDrawing.value) {
                        isDrawing.value = true;
                        currentLine.value = [pos];
                    } else {
                        currentLine.value.push(pos);
                    }
                    
                    draw();
                };

                const handleMouseMove = (e) => {
                    mousePos.value = getMousePos(e);
                    if (isDrawing.value) {
                        draw();
                    }
                };

                const handleMouseUp = (e) => {
                    // Keep drawing mode active
                };

                const handleDoubleClick = (e) => {
                    if (isDrawing.value && currentLine.value.length >= 2) {
                        finishDrawing();
                    }
                };

                const handleKeyDown = (e) => {
                    if (e.key === 'Escape') {
                        currentLine.value = [];
                        isDrawing.value = false;
                        draw();
                    }
                    if (e.key === 'Delete' && selectedWall.value) {
                        deleteSelected();
                    }
                    if (e.key === 'z' && (e.ctrlKey || e.metaKey)) {
                        undoLastPoint();
                    }
                };

                const selectWall = (e) => {
                    const pos = getMousePos(e);
                    const allWalls = [
                        ...walls.value.map(w => ({ ...w, type: 'external' })),
                        ...partitions.value.map(p => ({ ...p, type: 'internal' }))
                    ];

                    selectedWall.value = null;
                    
                    for (let wall of allWalls) {
                        for (let i = 0; i < wall.points.length - 1; i++) {
                            const dist = pointToLineDistance(
                                pos,
                                wall.points[i],
                                wall.points[i + 1]
                            );
                            if (dist < 10) {
                                selectedWall.value = wall;
                                break;
                            }
                        }
                        if (selectedWall.value) break;
                    }
                    
                    draw();
                };

                const pointToLineDistance = (point, lineStart, lineEnd) => {
                    const A = point.x - lineStart.x;
                    const B = point.y - lineStart.y;
                    const C = lineEnd.x - lineStart.x;
                    const D = lineEnd.y - lineStart.y;

                    const dot = A * C + B * D;
                    const lenSq = C * C + D * D;
                    let param = -1;
                    
                    if (lenSq !== 0) param = dot / lenSq;

                    let xx, yy;

                    if (param < 0) {
                        xx = lineStart.x;
                        yy = lineStart.y;
                    } else if (param > 1) {
                        xx = lineEnd.x;
                        yy = lineEnd.y;
                    } else {
                        xx = lineStart.x + param * C;
                        yy = lineStart.y + param * D;
                    }

                    const dx = point.x - xx;
                    const dy = point.y - yy;
                    return Math.sqrt(dx * dx + dy * dy);
                };

                const finishDrawing = () => {
                    if (currentLine.value.length >= 2) {
                        const lineData = {
                            points: [...currentLine.value],
                            id: Date.now()
                        };

                        if (currentTool.value === 'external') {
                            walls.value.push(lineData);
                        } else {
                            partitions.value.push(lineData);
                        }

                        currentLine.value = [];
                        isDrawing.value = false;
                        draw();
                    }
                };

                const undoLastPoint = () => {
                    if (currentLine.value.length > 0) {
                        currentLine.value.pop();
                        if (currentLine.value.length === 0) {
                            isDrawing.value = false;
                        }
                        draw();
                    }
                };

                const deleteSelected = () => {
                    if (!selectedWall.value) return;

                    if (selectedWall.value.type === 'external') {
                        walls.value = walls.value.filter(w => w.id !== selectedWall.value.id);
                    } else {
                        partitions.value = partitions.value.filter(p => p.id !== selectedWall.value.id);
                    }

                    selectedWall.value = null;
                    draw();
                };

                const clearAll = () => {
                    if (confirm('Вы уверены, что хотите очистить весь план?')) {
                        walls.value = [];
                        partitions.value = [];
                        currentLine.value = [];
                        isDrawing.value = false;
                        selectedWall.value = null;
                        draw();
                    }
                };

                const exportPlan = () => {
                    const link = document.createElement('a');
                    link.download = 'floorplan.png';
                    link.href = canvas.value.toDataURL('image/png');
                    link.click();
                };

                watch(currentTool, () => {
                    selectedWall.value = null;
                    draw();
                });

                return {
                    canvas,
                    currentTool,
                    walls,
                    partitions,
                    handleMouseDown,
                    handleMouseMove,
                    handleMouseUp,
                    handleDoubleClick,
                    finishDrawing,
                    undoLastPoint,
                    clearAll,
                    exportPlan
                };
            }
        }).mount('#app');
    </script>
